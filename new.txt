# 1. Clone the repo
git clone https://github.com/SomethingForWork/tests.git

# 2. Convert ASCII key to binary format and move to system keyrings
gpg --dearmor tests/server-6.0.asc | sudo tee /usr/share/keyrings/mongodb-server-6.0.gpg > /dev/null

# 3. Register the MongoDB 6.0 repo
echo "deb [signed-by=/usr/share/keyrings/mongodb-server-6.0.gpg] https://repo.mongodb.org/apt/ubuntu jammy/mongodb-org/6.0 multiverse" | sudo tee /etc/apt/sources.list.d/mongodb-org-6.0.list

# 4. Update packages and install MongoDB
sudo apt update
sudo apt install -y mongodb-org

# 5. Enable and start the service
sudo systemctl enable mongod
sudo systemctl start mongod
